import React, { useRef, useState, useEffect } from 'react'
import {
  StyleSheet,
  View,
  Animated,
  Easing,
  Image,
  TouchableOpacity,
} from 'react-native'
import MapView from 'react-native-maps';
import Icon from 'react-native-vector-icons/Ionicons'

const MapScreen = ({navigation}) => {
  const markerLift = useRef(new Animated.Value(0)).current
  const rippleAnim = useRef(new Animated.Value(0)).current

  const [region, setRegion] = useState({
    latitude: 12.9716,
    longitude: 77.5946,
    latitudeDelta: 0.01,
    longitudeDelta: 0.01,
  })

  // 🔵 Continuous Ripple Animation
  useEffect(() => {
    Animated.loop(
      Animated.timing(rippleAnim, {
        toValue: 1,
        duration: 1500,
        easing: Easing.out(Easing.ease),
        useNativeDriver: true,
      })
    ).start()
  }, [])

  const onRegionChange = () => {
    Animated.spring(markerLift, {
      toValue: 1,
      friction: 5,
      tension: 100,
      useNativeDriver: true,
    }).start()
  }

  const onRegionChangeComplete = (newRegion) => {
    setRegion(newRegion)

    Animated.spring(markerLift, {
      toValue: 0,
      friction: 5,
      tension: 100,
      useNativeDriver: true,
    }).start()
  }

  const translateY = markerLift.interpolate({
    inputRange: [0, 1],
    outputRange: [0, -30],
  })

  const rippleScale = rippleAnim.interpolate({
    inputRange: [0, 1],
    outputRange: [0.3, 1.5],
  })

  const rippleOpacity = rippleAnim.interpolate({
    inputRange: [0, 1],
    outputRange: [0.6, 0],
  })

  return (
    <View style={styles.container}>
      <MapView
        style={styles.map}
        initialRegion={region}
        onRegionChange={onRegionChange}
        onRegionChangeComplete={onRegionChangeComplete}
        showsUserLocation
        showsMyLocationButton
      />
      
      <TouchableOpacity onPress={()=> navigation.goBack()} style={{position:'absolute', top:15, left:15}} >
        <Icon name="arrow-back" size={30} />
      </TouchableOpacity>

      {/* Center Marker */}
      <View style={styles.markerWrapper}>
        
        {/* Cyclone Ripple */}
        <Animated.View
          style={[
            styles.ripple,
            {
              transform: [{ scale: rippleScale }],
              opacity: rippleOpacity,
            },
          ]}
        />

        {/* Marker */}
        <Animated.View
          style={{
            transform: [{ translateY }],
          }}
        >
          <Image
            source={require('../../assets/pin.png')}
            style={{ width: 38, height: 55 }}
          />
        </Animated.View>
      </View>
    </View>
  )
}

export default MapScreen

const styles = StyleSheet.create({
  container: { flex: 1 },
  map: { flex: 1 },

  markerWrapper: {
    position: 'absolute',
    top: '50%',
    left: '50%',
    alignItems: 'center',
    justifyContent: 'center',
    marginLeft: -22,
    marginTop: -45,
  },

  ripple: {
    position: 'absolute',
    width: 50,
    height: 50,
    top:27,
    borderRadius: 25,
    backgroundColor: 'rgb(44, 41, 1)',
  },
})
